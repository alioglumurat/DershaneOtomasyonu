PROCEDURES
ALTER procedure [dbo].[giderTipiEkle]
@giderAdi varchar (250)
as
begin
insert into gidertipleri(adi) values(@giderAdi)
end

ALTER procedure [dbo].[giderTipiGuncelle]
@giderID int,
@giderAdi varchar (250)
as
begin
Update gidertipleri set adi=@giderAdi where id=@giderID
end

ALTER procedure [dbo].[giderTipiListele]
as
begin
select * from gidertipleri
end

ALTER procedure [dbo].[giderEkle]
@giderAdi varchar (250),
@tutar money,
@aciklama varchar (500),
@dosya varchar (1000)

as
declare @giderID int
select @giderID = id from gidertipleri where adi = @giderAdi
begin
insert into gider(giderID,tutar,aciklama,dosya) values(@giderID,@tutar,@aciklama,@dosya)
end

ALTER procedure [dbo].[giderGuncelle]
@ID int,
@gadi varchar (250),
@aciklama varchar (500),
@tutar money,
@dosya varchar (1000)
as
declare @giderID int
select @giderID = id from gidertipleri where adi = @gadi;
begin
Update gider set aciklama = @aciklama, tutar = @tutar, dosya = @dosya, giderId = @giderID  where id = @ID
end

ALTER procedure [dbo].[giderSil]
@giderID int
as
begin
delete from gider where id = @giderID
end

ALTER procedure [dbo].[giderListele]
as
begin
select gd.id, gt.adi, gd.aciklama, gd.tutar, gd.dosya, gd.tarih from gider as gd left join gidertipleri as gt on gd.giderId = gt.id
end

ALTER procedure [dbo].[giderAdiCombo]
as
begin
select adi from gidertipleri
end

FULL TEXT CATALOG
ALTER procedure [dbo].[giderArama]
@kelime varchar (250)
as
begin
select gd.id, gt.adi, gd.aciklama, gd.tutar, gd.dosya, gd.tarih from gider as gd left join gidertipleri as gt on gd.giderId = gt.id where contains(aciklama,@kelime)
end
